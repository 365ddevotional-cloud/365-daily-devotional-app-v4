STEP 5 — Apply Customization to Scripture

const verse = document.querySelector(".cardScripture");

const sizeSlider = document.getElementById("fontSizeSlider");
const colorPicker = document.getElementById("textColorPicker");
const outlineToggle = document.getElementById("textOutline");

sizeSlider.oninput = () => {
  verse.style.fontSize = sizeSlider.value + "px";
  document.getElementById("fontSizeValue").innerText = sizeSlider.value + "px";
};

colorPicker.oninput = () => {
  verse.style.color = colorPicker.value;
};

document.getElementById("boldBtn").onclick = () => {
  verse.style.fontWeight = "800";
};

document.getElementById("italicBtn").onclick = () => {
  verse.style.fontStyle = "italic";
};

outlineToggle.onchange = () => {
  if(outlineToggle.checked){
    verse.style.webkitTextStroke = "1px black";
  } else {
    verse.style.webkitTextStroke = "0px";
  }
};
